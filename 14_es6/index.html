Modern JavaScript (ES6+)

1. let & const

În ES6, folosim let și const în loc de var:

let nume = "Ana";
const PI = 3.14;
nume = "Maria"; // OK
PI = 3.1415; // ❌ Eroare - const nu poate fi reasignat

2. Template literals

Permite interpolarea ușoară a valorilor:

const nume = "Ion";
console.log(`Salut, ${nume}!`); // Salut, Ion!

3. Arrow functions

const suma = (a, b) => a + b;
console.log(suma(2, 3)); // 5

4. Destructuring

const persoana = { nume: "Andrei", varsta: 25 };
const { nume, varsta } = persoana;
console.log(nume, varsta);

5. Spread & Rest

const arr1 = [1, 2, 3];
const arr2 = [...arr1, 4, 5];
console.log(arr2); // [1,2,3,4,5]

const suma = (...numere) => numere.reduce((a,b) => a+b, 0);
console.log(suma(1,2,3)); // 6

6. Default parameters

function salut(nume = "Utilizator") {
  console.log(`Salut, ${nume}!`);
}
salut(); // Salut, Utilizator!

7. Array methods: map, filter, reduce

const numere = [1, 2, 3, 4, 5];

const dublate = numere.map(n => n * 2);
const pare = numere.filter(n => n % 2 === 0);
const suma = numere.reduce((a, b) => a + b, 0);

console.log(dublate, pare, suma);

8. Module system (import/export)

// math.js
export const aduna = (a, b) => a + b;

// main.js
import { aduna } from './math.js';
console.log(aduna(2, 3));

9. Promises

const promisiune = new Promise((resolve, reject) => {
  setTimeout(() => resolve("Date primite!"), 1000);
});

promisiune.then(msg => console.log(msg));

10. Async/Await

async function preiaDate() {
  const raspuns = await fetch("https://jsonplaceholder.typicode.com/users");
  const date = await raspuns.json();
  console.log(date);
}
preiaDate();

Workshop: Manipulare date JSON

Exemplu de prelucrare date JSON cu metode ES6:

const utilizatori = [
  { nume: "Ana", varsta: 23 },
  { nume: "Mihai", varsta: 30 },
  { nume: "Ioana", varsta: 27 },
];

const peste25 = utilizatori.filter(u => u.varsta > 25);
console.log(peste25);
